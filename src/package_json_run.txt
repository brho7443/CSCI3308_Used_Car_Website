To run without debugging:
"scripts": {
    "prestart": "npm install && npm audit fix --force",
    "start": "nodemon index.js"
}

To run with debugging and exit:
"scripts": {
    "prestart": "npm install",
    "start": "node index.js",
    "dev": "nodemon index.js",
    "test": "mocha",
    "testandrun": "npm run prestart && npm run test"
}
